### 基本rpc
```go
type HelloService struct {}

func (p *HelloService) Hello(request string, reply *string) error {
	fmt.Println(request)
	*reply = "hello:" + request
	return nil
}

func TestTcpRpc(t *testing.T) {
	rpc.RegisterName("HelloService", new(HelloService))

	listener, err := net.Listen("tcp", ":1234")
	if err != nil {
		log.Fatal("ListenTCP error:", err)
	}
	for {
		conn, err := listener.Accept()
		if err != nil {
			log.Fatal("Accept error:", err)
		}
		fmt.Println(conn.RemoteAddr())
		go rpc.ServeConn(conn)
	}

}

func TestTcpRpcClient(t *testing.T) {
	client, err := rpc.Dial("tcp", "localhost:1234")
	if err != nil {
		log.Fatal("dialing:", err)
	}

	var reply string
	err = client.Call("HelloService.Hello", "golang", &reply)
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println(reply)
}
```

### 基于json的rpc服务
当我们需要跨语言调用时，采用json格式进行通信
```go
type HelloService struct {}

func (p *HelloService) Hello(request string, reply *string) error {
	fmt.Println(request)
	*reply = "hello:" + request
	return nil
}
func TestRpcJson(t *testing.T) {

	listener, err := net.Listen("tcp", ":1234")
	if err != nil {
		log.Fatal("ListenTCP error:", err)
	}

	for {
		conn, err := listener.Accept()

			log.Fatal("Accept error:", err)
		}

}
```

### 基于http的rpc服务
```go
func TestRpcHttp(t *testing.T) {
	rpc.RegisterName("HelloService", new(HelloService))

	http.HandleFunc("/jsonrpc", func(w http.ResponseWriter, r *http.Request) {
		var conn io.ReadWriteCloser = struct {
			io.Writer
			io.ReadCloser
		}{
			ReadCloser: r.Body,
			Writer:     w,
		}

		rpc.ServeRequest(jsonrpc.NewServerCodec(conn))
	})

	http.ListenAndServe(":1234", nil)
	time.Sleep(time.Minute * 10)
}
```

### 上下文信息
首先我们将连接信息作为Service成员
```go 
type HelloService struct {
	conn net.Conn
}
```
为每个连接单独启动rpc服务
```go
func main() {
	listener, err := net.Listen("tcp", ":1234")
	if err != nil {
		log.Fatal("ListenTCP error:", err)
	}

	for {
		conn, err := listener.Accept()
		if err != nil {
			log.Fatal("Accept error:", err)
		}

		go func() {
			defer conn.Close()

			p := rpc.NewServer()
			p.Register(&HelloService{conn: conn})
			p.ServeConn(conn)
		} ()
	}
}
```
服务处理中则可以获取到连接信息
```go
func (p *HelloService) Hello(request string, reply *string) error {
	*reply = "hello:" + request + ", from" + p.conn.RemoteAddr().String()
	return nil
}
```
