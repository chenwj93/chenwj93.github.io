> 库存 go-设计模式

- 观察者接口包含一个通知方法
- 被观察者维护一个观察者链表
- 观察者向被观察者订阅
- 被观察者在自身状态改变时

```go
package design_pattern

import (
	"fmt"
	"testing"
)

type IObserver interface {
	Notify()
}

type stat struct {
	observerList []IObserver
	status int
}

func (s *stat) setStatus(st int)  {
	s.status = st
	fmt.Println(s.observerList)
	for index := range s.observerList {
		s.observerList[index].Notify()
	}
}

func (s *stat) subscribe(observer IObserver)  {
	s.observerList = append(s.observerList, observer)
}

type observer1 struct {
	status *stat
}

func (o *observer1) SetStat(s *stat)  {
	o.status = s
	o.status.subscribe(o)
}

func (o *observer1) Notify()  {
	fmt.Println("observer 1", o.status.status)
}

type observer2 struct {
	status *stat
}

func (o *observer2) SetStat(s *stat)  {
	o.status = s
	o.status.subscribe(o)
}

func (o *observer2) Notify()  {
	fmt.Println("observer 2", o.status.status)
}

type observer3 struct {
	status *stat
}

func (o *observer3) SetStat(s *stat)  {
	o.status = s
	o.status.subscribe(o)
}

func (o *observer3) Notify()  {
	fmt.Println("observer 3", o.status.status)
}

func TestObserver(t *testing.T) {
	status := stat{}
	human := observer1{}
	dog := observer2{}
	cat := observer3{}
	human.SetStatus(&status)
	dog.SetStatus(&status)
	cat.SetStatus(&status)

	status.setStatus(5)
	fmt.Println("==================")
	status.setStatus(10)
}
```

#### 附(uml脚本)
```
@startuml
interface IObserver {
    Notify()
}

class stat  {
	observerList []IObserver
	status int

    setStatus(st int)
    subscribe(observer IObserver)
}

class observer1 {
	status *stat

    SetStat(s *stat)
    Notify()
}

class observer2 {
	status *stat

    SetStat(s *stat)
    Notify()
}

class observer3 {
	status *stat

    SetStat(s *stat)
    Notify()
}

IObserver <|..observer1
IObserver <|..observer2
IObserver <|..observer3
stat <..observer1
stat <..observer2
stat <..observer3
IObserver <.. stat

@enduml
```
